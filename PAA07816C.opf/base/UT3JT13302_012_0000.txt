༄༅། །ལམ་འབྲས་ཨ་སེང་དོན་བསྡུས་ཞེས་བྱ་བ་བཞུགས་སོ།། ༈ །ན་མོ་གུ་རུ་བིརྺ་པཱ་ཡ། རྗེ་བཙུན་རིན་ཆེན་དམ་པའི་གསུང་གང་ཞིག །མུན་བྲལ་ཟླ་ཤར་འོད་ཟེར་གང་ཞིག་གིས། །བདག་ལྟ་བློ་ཡི་མུན་པ་བྲལ་གྱུར་ནས། །རང་གི་སེམས་ལ་སྣང་བ་ཅུང་ཟད་ཐོབ། །གྲོགས་མཆོག་དམ་པ་ཐོས་པའི་རྒྱུད་སྦྱངས་ཤིང༌། །བྱང་ཆུབ་ཆེད་དུ་སྒྲུབ་པའི་གོ་བགོས་ནས། །དམ་པའི་སྤྱན་སྔར་ལུང་མནོས་དམ་བཅས་པ། །མི་མཆོག་ཁྱོད་ཀྱི་བཀའ་སྩལ་ལྕི་བར་གྱུར། །ཐམས་ཅད་མཁྱེན་པའི་བཀའ་ནི་རྒྱ་མཚོ་བཞིན། །ཟབ་ཅིང་གཏིང་དཔག་དཀའ་ལ་ཐུག་པ་མེད། །རྡོ་རྗེ་ཐེག་པ་དགོངས་པ་དུ་མ་ཡིས། །གསུངས་ཤིང་རྡོ་རྗེའི་ཚིག་གིས་རྒྱས་བཏབ་པས། །ལུང་བསྟན་གྱུར་པ་ཀླུ་སྒྲུབ་ལ་སོགས་སམ། །མཁའ་འགྲོས་བྱིན་རླབས་བིརྺ་པ་ལ་སོགས། །གྲུབ་ཐོབ་རྣམས་ཀྱི་གདམས་ངག་ལ་བརྟེན་ནས། །རྗེས་འཇུག་སྐྱེ་བོའི་བཀའ་རྣམས་རྟོགས་པར་འགྱུར། །དོན་གྱི་གཙོ་བོ་བིརྺ་ལ་སོགས་པའི། །ལམ་གྱི་གསུང་ངག་ཁྱེད་ཀྱི་ཐུགས་ལ་མངའ། །འོན་ཀྱང་བཀའ་འཕྲིན་མི་འགོག་པ་ཡི་ཕྱིར། །རང་བློ་ལས་བྱུང་འདི་ཙམ་ཡི་གེར་བཀོད། །སྣང་བ་གསུམ་གྱི་ལམ་གྱིས་གཞི་བཅས་ཏེ། །ཉམས་བླང་མ་ལུས་རྒྱུད་གསུམ་དག་གིས་ཤེས། །ཚད་མ་སྙན་རྒྱུད་བཞི་ཡིས་སྤྲོས་པ་གཅོད། །རྟེན་འབྲེལ་ལྔ་ཡིས་ལམ་གྱི་ས་མཚམས་གཅོད། །འདུ་ལུགས་གསུམ་དང་ཕྱོགས་མེད་བདུན་དག་གིས། །རྣམ་པར་ཕྱེ་བས་སྒྲོ་སྐུར་བྲལ་བར་མཛོད། །སེམས་འཛིན་གནས་ལུགས་རྣམ་གསུམ་རྟོགས་པ་ཡིས། །ཏིང་ངེ་འཛིན་གྱི་ཉེ་རྒྱུ་རྟོགས་པར་བྱ། །རང་ལུས་དག་ལ་ཉམས་གསུམ་སྐྱེ་བ་ཡིས། །གང་ཡང་ཏིང་ངེ་འཛིན་དུ་མཁྱེན་པར་མཛོད། །དྲོད་གསུམ་རིམ་པ་ཡིན་མིན་སྐྱེ་བ་ཡིས། །གང་ཡིན་བྱུང་རྒྱལ་དག་ཏུ་བཏང་བར་མཛོད། །དེད་དཔོན་ལྟ་བུའི་ཁྲིད་ལུགས་རྣམ་གསུམ་གྱིས། །ཏིང་འཛིན་ཉམས་ཀྱི་བྱེ་བྲག་རྟོགས་པར་མཛོད། །འདུས་བ་གསུམ་ལ་དཔེ་གསུམ་སྦྱར་བ་ཡིས། །ལམ་གྱི་ས་མཚམས་ཐེ་ཚོམ་མེད་པར་མཛོད། །རྟེན་འབྲེལ་རྣམ་གསུམ་རང་ལ་འཛོམ་པ་ཡིས། །རྒྱུ་རྐྱེན་གཞན་མེད་རེ་དོགས་བྲལ་བར་མཛོད། །ཐབས་དང་ཤེས་རབ་ཟུང་འཇུག་རྣལ་འབྱོར་པ། །གཉིས་ལ་བར་ཆད་འཇུག་སྟེ་གཅིག་ལ་མིན། །བསྲུང་བ་བརྒྱད་བརྒྱད་བཅུ་དྲུག་དག་ཏུ་སྟེ། །ཐབས་ལ་དད་པ་ལ་སོགས་རྣམ་པ་བརྒྱད། །ཤེས་རབ་སད་པ་རྣམ་པ་བཞི་ཡིས་བསྲུང༌། །ནང་གི་བདུད་ལ་དབང་བཞིའི་བརྡ་ཡིས་བཟློགས། །གཉིས་ལ་ཐུན་མོང་བསྲུང་བ་བཅུ་བཞི་སྟེ། །དང་པོ་ལམ་དུ་འཇུག་དང་ཞུགས་པ་དང༌། །སེམས་འཛིན་ཅུང་ཟད་བརྟན་དང་ཆེར་བརྟན་པའི། །བར་ཆད་རིམ་གྱིས་སེལ་བ་ཤེས་པར་བྱ། །རྩ་ཡི་ཟུག་ལ་ལ་དོར་ཤིན་ཏུ་གཅེས། །རླུང་གི་ཟུག་ལ་རྔན་དགས་བྱིན་ཏེ་བཟློག

།རླུང་རིད་པ་ལ་ལ་དོར་བྱིན་རླབས་དང༌། །བདེ་བ་སྒོམ་དང་རྣམ་པ་གསུམ་དུ་གསུངས། །ལྟ་སྒོམ་གཉིས་ལ་དུག་དབྱུང་བདུད་རྩི་བསྟེན། །ལ་དོར་མན་ངག་དྲུག་ཏུ་ཤེས་པར་བྱ། །ཐིག་ལེ་འབར་དང་གཡོ་དང་བརྟན་པ་དང༌། །རྣམ་གསུམ་དག་གིས་བདེ་སྟོང་མཆོག་ཏུ་འགྱུར། །འབར་བའི་ཚེ་ན་ཐིག་ལེའི་ཟུག་འབྱུང་སྟེ། །ཁ་སྦྱར་དག་ལ་བརྟན་པར་འགྱུར་བ་དང༌། །དུད་འགྲོའི་འགྱུར་བ་བཞི་ཡིས་བཟློག་པ་ཡིས། །ལུས་ལས་ཐིག་ལེ་གཡོ་བ་མེད་པར་བྱ། །ཀུན་ཏུ་ཁྱབ་ཅིང་བརྟེན་པར་བྱ་བའི་ཕྱིར། །ལུས་ལ་བཏུན་ཤིང་འཁོར་ལོ་བསྐོར་བ་དང༌། །ཡན་ལག་བཞི་ཡིས་རླུང་གི་ཞགས་པ་འཕང༌། །བུ་ཆུང་སྤྱོད་ལམ་དག་གིས་བརྟན་པར་འགྱུར། །རྗེ་བཙུན་དཔལ་ལྡན་བིརྺ་པ་ཡི་གཞུང༌། །དབང་བཞིའི་ལམ་ལ་ལྟ་གྲུབ་བརྒྱད་ཀྱིས་བསྡུས། །སྐྱོན་རྣམས་ཡོན་ཏན་དག་ཏུ་བླང་བ་དང༌། །བར་ཆད་དངོས་གྲུབ་དག་ཏུ་བླང་བར་གྱིས། །གང་ཡང་ཉམས་ཀྱི་སྣང་བར་མཁྱེན་བགྱིས་ན། །བདུད་དང་གོལ་ས་རང་གྲོལ་དག་ཏུ་འགྱུར། །ཐེ་ཚོམ་མེད་པར་བརྩོན་པ་ཆེར་མཛད་ན། །གྲུབ་མཐའ་སྙོགས་པར་འགྱུར་རོ་མི་མཆོག་ཁྱོད། །བདག་ལྟ་འདུ་བའི་ནང་ཞུགས་ནས། །སྒྲུབ་ལ་དད་ཀྱང་ལྷོད་པར་གྱུར། །རྗེ་བཙུན་བིརྺ་པ་ཡི་གཞུང༌། །ཡི་གེར་བཀོད་པ་དག་ཀྱང་མེད། །རང་གི་སེམས་ལ་སྣང་བ་ཙམ། །བཀོད་ན་དམ་པའི་བཀའ་ལས་འདས། །གྲོགས་ཀྱི་བཀའ་སྩལ་བཅག་གྱུར་ཀྱང༌། །མཁའ་འགྲོས་དཔང་མཛོད་བཟོད་པར་གསོལ། །རྗེ་བཙུན་རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་དཔལ་ས་སྐྱ་པ་ཆེན་པོ་ལ། མདོ་སྨད་ཀྱི་དགེ་སློང་རྡོ་རྗེ་འཛིན་པ་བྱང་ཆུབ་སེམས་དཔའ་ཨ་སེང་གིས་ཞུས་ཏེ་དོན་བསྡུས་པ་འདི་མཛད་དོ།། །།ས་མཱཔྟ་ཨི་ཐཱི།། །། ༈ བལ་ཡུལ་མཆོད་རྟེན་ཉེ་འདབས་ནས་རྒྱལ་ཡོངས་ས་ཆེན་ཞེས་པ་ལས། སློབ་བུ་ཐ་ཤལ་པ་གུ་རུས་སྤེལ་བའོ།།
