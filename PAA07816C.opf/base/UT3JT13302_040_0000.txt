༄༅། །ན་མོ་གུ་རུ་བཛྲ་ཡོ་གི་ནཱི་ཡེ། 

དེ་ལ་འདིར་དབང་ཐོབ་ཅིང༌། བྱིན་བརླབ་ཟབ་མོས་རྒྱུད་སྨིན་པའི་གང་ཟག་གིས། བསྡུས་པའི་ཚུལ་གྱིས་རྡོ་རྗེ་རྣལ་འབྱོར་མ་སྒོམ་པར་འདོད་པས། སྐྱབས་ཡུལ་གསལ་བཏབ་ནས། བདག་སོགས་འགྲོ་ཀུན་དེང་ནས་བྱང་ཆུབ་བར། །བླ་མ་དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་མཆི། །རྫོགས་བྱང་གོ་འཕང་ལ་བརྟེན་སེམས་ཅན་ཀུན། །བསྒྲལ་ཕྱིར་ཟབ་མོའི་རྣལ་འབྱོར་བསྒོམ་པར་བགྱི། །རང་གི་སྤྱི་བོར་པདྨ་ཟླ་བའི་སྟེང༌། །བླ་མ་རྡོར་སེམས་ཡབ་ཡུམ་གསལ་བ་ཡི། །སྐུ་ལས་བདུད་རྩིའི་རྒྱུན་བབས་སྡིག་ལྟུང་བཀྲུས། །དབང་བཞི་ཐོབ་ཅིང་གསང་གསུམ་བྱིན་བརླབས་གྱུར། །ཨོཾ་ཤྲཱི་བཛྲ་ཧེ་རུ་ཀ་ སོགས་ཡིག་བརྒྱ་ཅི་ནུས་བཟླས་མཐར། བླ་མ་ཐོག་མ་མེད་པའི་སེམས་དཔའ་ཡིས། །བདག་གི་སྔོན་བསགས་སྡིག་སྒྲིབ་ཉེས་ལྟུང་ཀུན། །བྱང་ཞིང་དག་པར་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། །སྤྱི་བོའི་རྡོར་སེམས་འོད་ཞུ་རང་ལ་ཐིམ། ། བླ་མའི་རྣལ་འབྱོར་ཐུན་མིན་ནི། རང་མདུན་ནམ་མཁར་སེང་ཁྲི་པད་ཟླའི་སྟེང༌། །རྩ་བའི་བླ་མ་སངས་རྒྱས་རྡོ་རྗེ་ཆོས། །མཐའ་སྐོར་པད་ཟླར་བརྒྱུད་པའི་བླ་མ་རྣམས། །དཔའ་བོ་རྡོ་རྗེ་ཆོས་ཀྱི་རྣམ་པར་བཞུགས། །མཐའ་སྐོར་སྐྱབས་གནས་རྒྱ་མཚོ་གསལ་བ་ཡི། །གནས་གསུམ་འོད་ཀྱིས་སྐྱབས་གནས་མ་ལུས་པ། །སྤྱན་དྲངས་བསྟིམ་པས་སྐྱབས་གནས་ཀུན་འདུས་གྱུར། །གང་གི་དྲིན་གྱིས་བདེ་ཆེན་ཉིད། །སྐད་ཅིག་ཉིད་ལ་འཆར་བ་གང༌། །བླ་མ་རིན་ཆེན་ལྟ་བུའི་སྐུ། །རྡོ་རྗེ་ཅན་ཞབས་པད་ལ་འདུད། །སྐྱབས་གནས་ཀུན་འདུས་བླ་མ་རིན་པོ་ཆེ། །དྲིན་ཅན་ཆོས་ཀྱི་རྗེ་ལ་གསོལ་བ་འདེབས། །མཉམ་མེད་བཀའ་དྲིན་ཅན་གྱི་ཐུགས་རྗེ་གཟིགས། །འདི་ཕྱི་བར་དོ་ཀུན་ཏུ་བྱིན་གྱིས་རློབས། ། ཞེན་པ་སྤང་བ་ནི། བདག་གི་ཉོན་མོངས་རང་གཟུགས་ལུས་ལོངས་སྤྱོད། །དགྲ་གཉེན་བར་མའི་སྣང་བ་ཇི་སྙེད་པ། །རྗེ་བཙུན་བླ་མ་འཁོར་དང་བཅས་ལ་འབུལ། །ཟུང་འཛིན་རྣམ་རྟོག་ཞི་བར་བྱིན་གྱིས་རློབས། །

མཆོད་པ་འབུལ་བ་ནི། ཕྱི་མཆོད་ཉེར་སྤྱོད་ནང་མཆོད་བདུད་རྩིའི་སྤྲིན། །གསང་བ་སྙོམས་འཇུག་བདེ་སྟོང་དེ་ཁོ་ན། །འཁོར་གསུམ་དམིགས་མེད་ཕྱག་རྒྱ་ཆེན་པོ་ཡིས། །རྩ་བརྒྱུད་དཀོན་མཆོག་རྒྱ་མཚོའི་ཚོགས་རྣམས་མཆོད། ། ཀུ་ས་ལིའི་ཚོགས་བསགས་ནི། རང་སེམས་རྣལ་འབྱོར་མ་དཔལ་མཁའ་ལ་ཐོན། །ཕྱག་གཡས་གྲི་གུག་ཕུང་པོའི་སྨིན་མཚམས་སུ། །རེག་པ་ཙམ་གྱིས་ཐོད་པ་ལིང་གིས་བཀོག །མདུན་དུ་ཐོད་པའི་སྒྱིད་བུ་གསུམ་གྱི་སྟེང༌། །ལེགས་བཀལ་སྟོང་གསུམ་ཁྱོན་དང་མཉམ་པར་གྱུར། །རང་ལུས་དུམ་བུར་གཏུབས་ཏེ་དེའི་ནང་བསྐྱུར། །ཡེ་ཤེས་མེ་འབར་ཐོད་པའི་ནང་རྫས་རྣམས། །ཞུ་ཞིང་ཁོལ་བའི་བདུད་རྩིའི་རྒྱ་མཚོར་གྱུར། །ཨོཾ་ཨཱ་ཧཱུྃ་ཧ་ཧོཿཧྲཱི། ཞེས་པའི་བྱིན་བརླབ། དྲིན་ཅན་རྩ་བའི་བླ་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྃ། ཞེས་ཚར་བདུན། བརྒྱུད་པའི་བླ་མ་རྣམས་ཀྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྃ། སྐྱབས་གནས་དཀོན་མཆོག་རྒྱ་མཚོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྃ། རིགས་དྲུག་ལྷར་གསལ་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྃ། བདུད་རྩི་མཐོང་རེག་མྱང་བས་སྒྲིབ་གཉིས་བྱང༌། །ཕུང་པོ་འཇའ་ལུས་སེམས་ཉིད་ཆོས་སྐུར་གྲོལ། །ས་གཞི་སྤོས་ཆུས་བྱུགས་ཤིང་མེ་ཏོག་བཀྲམ། །རི་རབ་གླིང་བཞི་ཉི་ཟླས་བརྒྱན་པ་འདི། །སངས་རྒྱས་ཞིང་དུ་དམིགས་ཏེ་ཕུལ་བ་ཡིས། །འགྲོ་ཀུན་རྣམ་དག་ཞིང་དུ་སྤྱོད་པར་ཤོག ། དབང་བླང་བ་ནི། བླ་མ་སྐྱབས་གནས་རྒྱ་མཚོའི་ཚོགས་རྣམས་ལ། །དབང་བཞི་རྫོགས་པར་གནང་ཞེས་གསོལ་བཏབ་པས། །རྩ་བརྒྱུད་བླ་མས་གནང་བ་ཞུས་ཐོབ་སྟེ། །གཡས་བསྐོར་མས་རིམ་རྩ་བའི་བླ་མར་ཐིམ། །བླ་མའི་གནས་གསུམ་འབྲུ་གསུམ་འོད་ཟེར་ཚོགས། །རིམ་དང་ཅིག་ཅར་བྱུང་ནས་རང་ལ་ཐིམ། ། བླ་མ་བསྟིམ་པ་ནི། རྩ་བའི་བླ་མར་རྩེ་གཅིག་གསོལ་བཏབ་པས། །དགྱེས་བཞིན་འོད་ཞུ་རང་ལ་ཐིམ་པ་ཡིས། །རང་སེམས་བྃ་ཡིག་སྤྲོས་བྲལ་དབྱིངས་སུ་ཐིམ། །དེ་ཡི་ངང་ལས་སྙིང་གར་ཨེ་ཨེ་ལས། །ཆོས་འབྱུང་ཟླ་དཀྱིལ་དབུས་སུ་བྃ་ཡིག་མཐར། །སྔགས་འཕྲེང་འཁོར་བའི་ཟེར་གྱིས་གཞི་ལུས་སྦྱངས། །འོད་ཀྱི་ཕུང་པོར་གྱུར་ལས་སྐད་ཅིག་གིས། །རང་ཉིད་པད་ཉི་དུས་མཚན་འཇིགས་བྱེད་སྟེང༌། །རྣམ་སྣང་རིགས་བདག་བཅས་པའི་རྣལ་འབྱོར་མ། །དམར་གསལ་སྤྱན་གསུམ་ནམ་མཁའི་དབྱིངས་སུ་གཟིགས། །ཕྱག་གཡས་གྲི་གུག་ཐུར་དུ་བརྐྱངས་ནས་འཛིན། །གཡོན་པས་ཐོད་ཁྲག་སྟེང་ཕྱོགས་བཟུང་བས་གསོལ། །ཕྲག་པ་གཡོན་པར་རྡོ་རྗེའི་ཁ་ཊྭཱཾ་བསྟེན། །ཐོད་སྐམ་དབུ་རྒྱན་དོ་ཤལ་ཕྱག་རྒྱས་སྤྲས། །ཞབས་ཟུང་གཡས་བརྐྱངས་གཡོན་བསྐུམ་དོར་ཐབས་བཞེངས། །བདེ་ཆེན་ཉམས་འབར་ཡེ་ཤེས་མེ་དབུས་རོལ། །ཐུགས་སྲོག་འོད་ཀྱི་སྣོད་བཅུད་དངོས་འཛིན་སྦྱངས། །རྡོ་རྗེ་བཙུན་མོའི་ཞིང་སྐུར་ལམ་གྱིས་གསལ། །

འབར་བའི་ཕྱག་རྒྱ་དང་བཅས། ཕཻཾ། སླར་ཡང་ཐུགས་ཀའི་བྃ་ལས་འོད་འཕྲོས་པས། །ལྷན་སྐྱེས་ཞིང་སྔགས་རྣལ་འབྱོར་མ་དཔལ་ཀུན། །ཚུར་འདུས་ཛཿཧཱུྃ་བཾ་ཧོཿས་རང་ལ་ཐིམ། །དམ་ཚིག་ཡེ་ཤེས་གཉིས་སུ་མེད་པར་གྱུར། །ཡེ་ཤེས་ཞུགས་པའི་རྐྱེན་གྱིས་སྙིང་ཁའི་བྃ། །འོད་ཞུ་དབུས་སུ་མཁའ་སྤྱོད་དམར་པོ་དང༌། །བྱང་ནས་གཡོན་བསྐོར་ལྗང་དམར་སེར་དང་དཀར། །མཐའ་བསྐོར་སྔགས་ཕྲེང་སུམ་ཅུ་སོ་གཉིས་རྣམས། །རྣལ་འབྱོར་མ་དཔལ་སུམ་ཅུ་སོ་གཉིས་གྱུར། །ཨོཾ་ཡོ་ག་ཤུདྡྷཿསརྦ་དྷརྨཱཿཡོ་ག་ཤུདྡྷོ྅ཧཾ། ལུས་གནས་སོ་སོར་ཟླ་བའི་དཀྱིལ་འཁོར་སྟེང༌། །ངོ་བོ་ལྷ་ལ་རྣམ་པ་ཡི་གེའི་གཟུགས། །ཨོཾ་བྃ་ཧྃ་ཡོཾ་ཧྲིཾ་མོཾ་ཧྲིཾ་ཧྲིཾ་དང༌། །ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་རྣམས་ཀྱིས་མཚན་པ་ཡི། །ཐུགས་ཀའི་ལུས་དཀྱིལ་གཙོ་མོའི་སྔགས་ཕྲེང་ལས། །འཕྲོས་པའི་འོད་ཀྱིས་འཕགས་མཆོད་འགྲོ་སྒྲིབ་སྦྱངས། །ཕོ་ཉ་གསུམ་གྱི་བྱིན་རླབས་ནུས་མཐུ་ཀུན། །བསྡུས་ཐིམ་མཁའ་སྤྱོད་དངོས་གྲུབ་ཐོབ་པར་གྱུར། །ཨོཾ་ཨོཾ་ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ། བཛྲ་ཝརྞ་ནཱི་ཡེ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ་སྭཱཧཱ། དམ་བཅའི་གྲངས་ལས་མི་ཉུང་བ་བཟླ། སྤྲོ་ན་ཡིད་བཟླས་སོགས་ཀྱི་མཐར། བླ་མ་རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མས། །འགྲོ་ཀུན་མཁའ་སྤྱོད་འཁྲིད་པར་མཛད་དུ་གསོལ། །ཁམས་གསུམ་འོད་ཞུ་རང་དང་རང་ཉིད་ཀྱང༌། །ནཱ་དའི་བར་ཐིམ་འོད་གསལ་དབྱིངས་སུ་ཡལ། །སླར་ཡང་རྒྱན་དང་ཕྱག་མཚན་སྤངས་པ་ཡི། །ལྷ་སྐུ་གོ་ཆས་མཚན་པར་གསལ་བའི་ངང༌། །སྣང་གྲགས་རྟོག་ཚོགས་ལྷ་སྔགས་ཆོས་ཀྱི་སྐུ། །དགའ་བ་གསུམ་དང་དབྱེར་མེད་འདྲེས་པར་གྱུར། །དགེ་བའི་རྩ་བ་འདི་ཡིས་བདག་གཞན་ཀུན། །མཁའ་སྤྱོད་ཆེན་པོའི་གོ་འཕང་འགྲུབ་རྒྱུར་ནས། །སྲིད་དང་ཞི་བའི་རྒུད་པ་ཀུན་སེལ་ཞིང༌། །དོན་གཉིས་ལྷུན་གྱིས་གྲུབ་པའི་བཀྲ་ཤིས་ཤོག །འཇམ་མགོན་མཁྱེན་བརྩེའི་གསུང་སྒྲུབ་ཐབས་ཐུན་མོང་བ་ལ་ཐུན་མིན་གྱི་འདོན་ཆ་ཅུང་ཟད་སྐད་གཉིས་སྨྲ་བ་ངག་དབང་བསམ་གཏན་རྣམ་གཉིས་ཀྱི་ཐུགས་འདོད་ལྟར་སུ་བྷ་ཥི་ཏས་བྲིས་པའོ།།

།། ༈ བལ་ཡུལ་མཆོད་རྟེན་ཉེ་འདབས་ནས་རྒྱལ་ཡོངས་ས་ཆེན་ཞེས་པ་ལས། སློབ་བུ་ཐ་ཤལ་པ་གུ་རུས་སྤེལ་བའོ།།
